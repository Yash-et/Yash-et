class TimeSeriesTransformer(nn.Module):
    """
    Custom transformer tuned for financial time series.
    Multi-head attention across temporal sequences with
    market-periodicity-aware positional encoding.
    """
    def __init__(self, d_model=512, n_heads=8, seq_len=60, dropout=0.1):
        super().__init__()
        self.temporal_embed  = TemporalPositionalEncoding(d_model, seq_len)
        self.encoder_stack   = nn.TransformerEncoder(
            nn.TransformerEncoderLayer(d_model, n_heads, dim_feedforward=2048,
                                       dropout=dropout, batch_first=True),
            num_layers=6
        )
        self.forecast_head   = nn.Sequential(
            nn.LayerNorm(d_model),
            nn.Linear(d_model, 128),
            nn.GELU(),
            nn.Linear(128, 1)
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x = self.temporal_embed(x)           # inject time-awareness
        x = self.encoder_stack(x)            # attend across timesteps
        return self.forecast_head(x[:, -1])  # predict next close
